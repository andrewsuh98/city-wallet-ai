# Data Models

This document defines every data type shared between frontend and backend. It is the single source of truth for the API contract. Backend uses Pydantic (Python), frontend uses TypeScript interfaces, and the database uses SQLite.

---

## Pydantic Models (Python)

All models live in `backend/models.py`.

### Context Types

```python
from pydantic import BaseModel
from enum import Enum
from datetime import datetime
from typing import Optional


class WeatherCondition(str, Enum):
    CLEAR = "clear"
    CLOUDY = "cloudy"
    RAIN = "rain"
    SNOW = "snow"
    STORM = "storm"


class WeatherData(BaseModel):
    condition: WeatherCondition
    temp_celsius: float
    humidity: int
    description: str  # e.g. "light rain"


class EventData(BaseModel):
    id: str
    name: str
    venue: str
    category: str  # "sports", "music", "festival", "market"
    start_time: datetime
    end_time: Optional[datetime]
    latitude: float
    longitude: float
    expected_attendance: Optional[int]


class TransactionDensity(BaseModel):
    merchant_id: str
    current_hour_txns: int
    avg_hour_txns: float
    density_ratio: float  # current / average. <0.5 = quiet, >1.5 = busy
    trend: str  # "quiet", "normal", "busy", "surging"


class UserLocation(BaseModel):
    latitude: float
    longitude: float
    accuracy_meters: Optional[float]


class ContextState(BaseModel):
    """The composite context object. Central data structure of the system."""
    timestamp: datetime
    location: UserLocation
    weather: WeatherData
    time_of_day: str       # "early_morning", "morning", "lunch", "afternoon", "evening", "night"
    day_of_week: str       # "monday" through "sunday"
    is_weekend: bool
    nearby_events: list[EventData]
    merchant_densities: list[TransactionDensity]
    context_tags: list[str]  # derived: ["rainy", "lunch_hour", "quiet_cafes"]
    urgency_score: float     # 0.0 to 1.0
```

### Merchant Types

```python
class MerchantCategory(str, Enum):
    CAFE = "cafe"
    RESTAURANT = "restaurant"
    RETAIL = "retail"
    BAKERY = "bakery"
    BAR = "bar"
    BOOKSTORE = "bookstore"
    GROCERY = "grocery"
    FITNESS = "fitness"


class MerchantRule(BaseModel):
    id: Optional[str]
    merchant_id: str
    max_discount_percent: int       # e.g. 20
    min_discount_percent: int       # e.g. 5
    active_hours_start: str         # "14:00"
    active_hours_end: str           # "17:00"
    trigger_conditions: list[str]   # ["quiet_period", "rainy", "weekday"]
    goal: str                       # "fill quiet afternoon hours"
    max_offers_per_day: int
    offer_type: str                 # "percentage_discount", "fixed_amount", "bogo", "free_item"
    budget_daily_usd: Optional[float]


class Merchant(BaseModel):
    id: str
    name: str
    category: MerchantCategory
    description: str
    latitude: float
    longitude: float
    address: str
    image_url: Optional[str]
    rules: list[MerchantRule]
```

### Offer Types

```python
class OfferStatus(str, Enum):
    ACTIVE = "active"
    ACCEPTED = "accepted"
    DECLINED = "declined"
    EXPIRED = "expired"
    REDEEMED = "redeemed"


class OfferStyle(BaseModel):
    """GenUI styling hints generated by Claude."""
    background_gradient: list[str]  # e.g. ["#FF6B35", "#F7C59F"]
    emoji: str                       # e.g. "coffee emoji"
    tone: str                        # "warm", "urgent", "playful", "sophisticated"
    headline_style: str              # "emotional", "factual"


class Offer(BaseModel):
    id: str
    merchant_id: str
    merchant_name: str
    merchant_category: MerchantCategory
    # Generated content
    headline: str          # max ~8 words
    subtext: str           # max ~15 words
    description: str       # 2-3 sentences for detail view
    discount_value: str    # "15%" or "$3 off"
    discount_type: str     # percentage_discount, fixed_amount, free_item, bogo
    # Context
    context_tags: list[str]
    why_now: str           # "It's raining and this cafe is quiet right now"
    # Timing
    created_at: datetime
    expires_at: datetime
    # Style (GenUI)
    style: OfferStyle
    # State
    status: OfferStatus
    distance_meters: Optional[float]
    # Redemption
    redemption_token: Optional[str]
```

## SQLite Schema

All tables are created in `backend/database.py`. Use `aiosqlite` for async access.

```sql
CREATE TABLE IF NOT EXISTS merchants (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    category TEXT NOT NULL,
    description TEXT,
    latitude REAL NOT NULL,
    longitude REAL NOT NULL,
    address TEXT,
    image_url TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS merchant_rules (
    id TEXT PRIMARY KEY,
    merchant_id TEXT NOT NULL REFERENCES merchants(id),
    max_discount_percent INTEGER DEFAULT 20,
    min_discount_percent INTEGER DEFAULT 5,
    active_hours_start TEXT DEFAULT '00:00',
    active_hours_end TEXT DEFAULT '23:59',
    trigger_conditions TEXT,    -- JSON array stored as text
    goal TEXT,
    max_offers_per_day INTEGER DEFAULT 10,
    offer_type TEXT DEFAULT 'percentage_discount',
    budget_daily_usd REAL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS offers (
    id TEXT PRIMARY KEY,
    merchant_id TEXT NOT NULL REFERENCES merchants(id),
    merchant_name TEXT NOT NULL,
    merchant_category TEXT NOT NULL,
    headline TEXT NOT NULL,
    subtext TEXT NOT NULL,
    description TEXT,
    discount_value TEXT,
    discount_type TEXT,
    context_tags TEXT,          -- JSON array as text
    why_now TEXT,
    style TEXT,                 -- JSON object as text (OfferStyle)
    status TEXT DEFAULT 'active',
    distance_meters REAL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expires_at TIMESTAMP,
    redemption_token TEXT,
    user_session_id TEXT
);

CREATE TABLE IF NOT EXISTS redemptions (
    id TEXT PRIMARY KEY,
    offer_id TEXT NOT NULL REFERENCES offers(id),
    token TEXT NOT NULL,
    redeemed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    cashback_amount REAL
);

-- Index for fast lookups
CREATE INDEX IF NOT EXISTS idx_offers_session ON offers(user_session_id);
CREATE INDEX IF NOT EXISTS idx_offers_merchant ON offers(merchant_id);
CREATE INDEX IF NOT EXISTS idx_offers_status ON offers(status);
CREATE INDEX IF NOT EXISTS idx_redemptions_token ON redemptions(token);
```

## Type Mapping Reference

| Python (Pydantic) | SQLite | TypeScript |
|-------------------|--------|------------|
| `str` | `TEXT` | `string` |
| `int` | `INTEGER` | `number` |
| `float` | `REAL` | `number` |
| `bool` | `INTEGER` (0/1) | `boolean` |
| `datetime` | `TIMESTAMP` (TEXT) | `string` (ISO) |
| `Optional[T]` | nullable | `T \| null` |
| `list[str]` | `TEXT` (JSON) | `string[]` |
| `Enum` | `TEXT` | union type |
| nested `BaseModel` | `TEXT` (JSON) | interface |

### JSON-in-SQLite Convention

Lists and nested objects are stored as JSON-encoded text in SQLite. When reading from the database, parse them with `json.loads()`. When writing, serialize with `json.dumps()`. The Pydantic model handles this automatically via custom validators if needed, but for simplicity the routers can handle serialization at the DB boundary.
